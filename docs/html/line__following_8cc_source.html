00161     {
00162         TRACE("correct_steering()");
00163 
00164         // Calculate available headroom for applying a forward
00165         // correction
00166         unsigned short int headroom = MOTOR_MAX_SPEED - this->_speed;
00167 
00168         // Switch based on error direction
00169         if(_left_error) {
00170             DEBUG("Correcting a left error");
00171 
00172             // Calculate the required differential correction
00173             unsigned short int correction = static_cast<unsigned short int>(
00174                 static_cast<double>(_left_error) * this->_integral_gain);
00175             
00176             DEBUG("Pre-cap correction: " << correction);
00177 
00178             // Cap correction
00179             correction = cap_correction(correction);
00180 
00181             DEBUG("Post-cap correction: " << correction);
00182 
00183             // Apply as much correction as possible to the left
00184             // wheel forwards and whatever is left to right backwards
00185             if(headroom >= correction) {
00186                 this->_hal->motor_left_forward(this->_speed + correction);
00187                 this->_hal->motor_right_forward(this->_speed);
00188             } else {
00189                 this->_hal->motor_left_forward(this->_speed + headroom);
00190                 this->_hal->motor_right_forward(
00191                     this->_speed - (correction - headroom));
00192             }
00193 
00194         } else if(_right_error) {
00195             DEBUG("Correcting a right error");
00196 
00197             // Calculate the required differential correction
00198             unsigned short int correction = static_cast<unsigned short int>(
00199                 static_cast<double>(_right_error) * this->_integral_gain);
00200             
00201             DEBUG("Pre-cap correction: " << correction);
00202 
00203             // Cap correction
00204             correction = cap_correction(correction);
00205 
00206             DEBUG("Post-cap correction: " << correction);
00207 
00208             // Apply as much correction as possible to the right
00209             // wheel forwards and whatever is right to left backwards
00210             if(headroom >= correction) {
00211                 this->_hal->motor_right_forward(this->_speed + correction);
00212                 this->_hal->motor_left_forward(this->_speed);
00213             } else {
00214                 this->_hal->motor_right_forward(this->_speed + headroom);
