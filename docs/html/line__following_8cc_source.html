00157     {
00158         TRACE("correct_steering()");
00159 
00160         // Calculate available headroom for applying a forward
00161         // correction
00162         unsigned short int headroom = MOTOR_MAX_SPEED - this->_speed;
00163 
00164         // Switch based on error direction
00165         if(_left_error) {
00166             DEBUG("Correcting a left error");
00167 
00168             // Calculate the required differential correction
00169             unsigned short int correction = static_cast<unsigned short int>(
00170                 static_cast<double>(_left_error) * INTEGRAL_GAIN);
00171             
00172             DEBUG("Pre-cap correction: " << correction);
00173 
00174             // Cap correction
00175             correction = cap_correction(correction);
00176 
00177             DEBUG("Post-cap correction: " << correction);
00178 
00179             // Apply as much correction as possible to the left
00180             // wheel forwards and whatever is left to right backwards
00181             if(headroom >= correction) {
00182                 this->_hal->motor_left_forward(this->_speed + correction);
00183                 this->_hal->motor_right_forward(this->_speed);
00184             } else {
00185                 this->_hal->motor_left_forward(this->_speed + headroom);
00186                 this->_hal->motor_right_forward(
00187                     this->_speed - (correction - headroom));
00188             }
00189 
00190         } else if(_right_error) {
00191             DEBUG("Correcting a right error");
00192 
00193             // Calculate the required differential correction
00194             unsigned short int correction = static_cast<unsigned short int>(
00195                 static_cast<double>(_right_error) * INTEGRAL_GAIN);
00196             
00197             DEBUG("Pre-cap correction: " << correction);
00198 
00199             // Cap correction
00200             correction = cap_correction(correction);
00201 
00202             DEBUG("Post-cap correction: " << correction);
00203 
00204             // Apply as much correction as possible to the right
00205             // wheel forwards and whatever is right to left backwards
00206             if(headroom >= correction) {
00207                 this->_hal->motor_right_forward(this->_speed + correction);
00208                 this->_hal->motor_left_forward(this->_speed);

00394     {
00395         TRACE("junction_status()");
00396 
00397         // Read the state of the IR sensors from hal
00398         const LineSensors s = _hal->line_following_sensors();
00399 
00400         // If the inner sensors do not detect a line, it implies we
00401         // are not really at a junction anyway, but maybe have drifted
00402         // far enough to a side that we think the main line is a junction.
00403         // Therefore, return early in this case.
00404         //
00405         // Otherwise, we check the outer sensors for junctions.
00406         if(s.line_left == NO_LINE || s.line_right == NO_LINE) {
00407             DEBUG("Not looking at edge sensors as inner sensors are NO_LINE");
00408             return NO_TURNS_FOUND;
00409         } else if(s.outer_left == LINE && s.outer_right == LINE) {
00410             INFO("Found both turns");
00411             return BOTH_TURNS_FOUND;
00412         } else if(s.outer_left == LINE) {
00413             INFO("Found left turn");
00414             return LEFT_TURN_FOUND;
00415         }  else if(s.outer_right == LINE) {
00416             INFO("Found right turn");
00417             return RIGHT_TURN_FOUND;
00418         } else {
00419             DEBUG("No turns found");
00420             return NO_TURNS_FOUND;
00421         }
