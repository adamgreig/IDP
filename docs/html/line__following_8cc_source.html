00196         // Calculate available headroom for applying a forward
00197         // correction
00198         unsigned short int headroom = MOTOR_MAX_SPEED - this->_speed;
00199 
00200         // Switch based on error direction
00201         if(_left_error) {
00202             DEBUG("Correcting a left error");
00203 
00204             // Calculate the required differential correction
00205             unsigned short int correction = static_cast<unsigned short int>(
00206                 static_cast<double>(_left_error) * INTEGRAL_GAIN);
00207             
00208             DEBUG("Pre-cap correction: " << correction);
00209 
00210             // Cap correction
00211             correction = cap_correction(correction);
00212 
00213             DEBUG("Post-cap correction: " << correction);
00214 
00215             // Apply as much correction as possible to the left
00216             // wheel forwards and whatever is left to right backwards
00217             if(headroom >= correction) {
00218                 this->_hal->motor_left_forward(this->_speed + correction);
00219                 this->_hal->motor_right_forward(this->_speed);
00220             } else {

00222                 this->_hal->motor_right_forward(
00223                     this->_speed - (correction - headroom));
